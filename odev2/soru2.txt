import java.io.*;
import java.util.*;
import java.net.*;

public class URL
{
    public static void main(String[] args)
    {
        try
        {
            java.net.URL url=new java.net.URL("https://tr.wikipedia.org/wiki/Java_(programlama_dili)");

            System.out.println("Protokol: "+url.getProtocol());
            System.out.println("Sunucu Adı: "+url.getHost());
            System.out.println("Port Sayısı: "+url.getPort());
            System.out.println("Varsayılan Port Sayısı: "+url.getDefaultPort());
            System.out.println("Yol: "+url.getPath());
            System.out.println("Dosya: "+url.getFile());
        }
        catch(Exception e)
        {
            e.printStackTrace();
        }
    }
}
